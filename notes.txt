// создание индекса

db.courts.createIndex(
{ COURTNAME: "text", COURTADDRESS: "text" },
{ default_language: "russian" },
{name: "RegionIndex"},
   {
     weights: {
       COURTADDRESS: 10,
       COURTNAME: 5
     } )

// показать индексы
db.courts.getIndexes()

// поиск по индексу
db.courts.find({$text: {$search: "Москва"}})
